01. 변수 : 데이터 저장소 

{
    var x = 100;         //변수 x에 숫자 100을 저장
    var y = 200;         //변수 y에 숫자 200을 저장
    var z = "javascript";   //변수 z에 문자 javascrips를 저장

    document.write(x);
    document.write(y);
    document.write(z);
}

02. 변수 : 데이터 저장 + 데이터 변경

{
    let x = 100;
    let y = 200;
    let z = "javascripts";
    x = 300;    //변수 x의값이 100에서 300으로 변경
    y = 400;
    z = "jquery";

    document.write(x;
    document.write(y);
    document.write(z);
}

03. 변수 : 데이터 저장 + 데이터 변경 + 데이터 추가 

{
    let x = 100;
    let y = 200;
    let z = "javascripts";
    x += 300;    //x = x + 300; (대입연산자)
    y += 400;    //y = y + 400; 
    z += "jquery"; //z = z + "jquery";

    document.write(x);
    document.write(y);
    document.write(z);
}

05. 배열 : 데이터 저장 (여러 개의 데이터 저장) : 표현 방법1 

{

    document.write(arr[0]);
    document.write(arr[1]);
    document.write(arr[2]);
}

16. 객체 : 데이터 저장 (키와 값) - 표현 방법 8 (객체 속에 함수)

{
    const obj = {
        a: 100,
        b: [200, 300],
        c: "javascript",
        d: function( ) {     
            document.write("자바스크립트가 실행되었습니다.");       
        },      
        e: function( ) {         
            document.write( obj.c + "가 실행되었습니다.");      
        },      
        f: function( ) {     
            document.write( this.c + "가 실행되었습니다.");     
        },        
    }       
    
    document.write(obj.a);      
    document.write(obj.b);      
    document.write(obj.b[0]);       
    document.write(obj.b[1]);       
    document.write(obj.c);      
    obj.d( );        
    obj.e( );           
    obj.f( );            
}

17. 변수의 종류 : 지역변수, 전역변수

{
    let x = 100; //전역변수
    let y = 200;
    let z = "javascript";
    let a = 300;

    function func(){
        let x = 100; //함수안의 지역변수
        let y = 200;
        let z = "javascript";
        a = 400;  //전역변수 : let없이 a값 변경 = 전역변수
        y = 500;  //이미 함수안에서 let y을 선언해 지역변수 y값이 500으로 변경

        document.write(x);
        document.write(y);
        document.write(z);
        document.write(a);
    }
    func();
    
    document.write(x);
    document.write(y);
    document.write(z);
    document.write(a);
}

18. 변수의 변환 : 형 변환

{
    let x = 100;
    let y = "100";
        
    document.write("x: " + x + " : " + typeof(x));
    document.write("y: " + y + " : " + typeof(y));
        
    x = "" + x; //숫자100에 ""추가해서 문자로 형변환 한것
    document.write("x: " + x + " : " + typeof(x));
        
    y = Number(y); //Number라는 메서드를 이용한 글자100를 숫자로 형변환
    document.write("y: " + y + " : " + typeof(y));
        
    x = Number(x);
    document.write("x: " + x + " : " + typeof(x));
        
    y = String(y);  //string메서드 이용해서 변환
    document.write("y: " + y + " : " + typeof(y));
}

19. 변수의 유형 : 데이터 타입

{
    let num1 = 100;
    let str = "javascript"
    let fun = function(){};
    let arr = [];
    let obj = {};
    let boo = true;
    let nul;   //아무런 값이 없을 때
    let und = undefined;  //값이 정이 되어 있지 않을 때
    let sym = Symbol();
        
    document.write(typeof(num1));
    document.write(typeof(str));
    document.write(typeof(fun));
    document.write(typeof(arr));
    document.write(typeof(obj));
    document.write(typeof(boo));
    document.write(typeof(nul));
    document.write(typeof(und));
    document.write(typeof(sym));
}
