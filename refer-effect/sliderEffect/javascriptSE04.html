const sliderInner = document.querySelector(".slider-inner.slider-inner1")
const slider = sliderInner.querySelectorAll(".slider.slider1")
const sliderInner2 = document.querySelector(".slider-inner.slider-inner2")

let currentIndex = 0;
let currentIndex2 = 0;
let sliderCount = slider.length;
let sliderHeight = sliderInner.querySelector(".slider").offsetHeight;
let sliderClone = sliderInner2.firstElementChild.cloneNode(true);
sliderInner2.appendChild(sliderClone);

sliderInner.style.top = "0px";

setInterval(()=>{
    (currentIndex < sliderCount-1) ? currentIndex++ : currentIndex = 0;
    let sliderPosition = currentIndex * -sliderHeight + "px";
    
    sliderInner.style.transition = "all 0.3s"
    sliderInner.style.top = sliderPosition;

    currentIndex2++;
    sliderInner2.style.transition = "all 0.3s"
    sliderInner2.style.transform = "translateY("+-sliderHeight * currentIndex2+"px)";

    if( currentIndex2 == sliderCount ){
        setTimeout(() => {
            sliderInner2.style.transition = "0s";
            sliderInner2.style.transform = "translateY(0)";
        }, 300)
        currentIndex2 = 0;
    }
}, 3000);
